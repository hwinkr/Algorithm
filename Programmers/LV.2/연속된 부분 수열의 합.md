```python
def solution(sequence, k):
    start = end = len(sequence) - 1
    prefix_sum = sequence[-1]

    while True:
        if prefix_sum < k:
            start -= 1
            prefix_sum += sequence[start]
        elif prefix_sum > k:
            prefix_sum -= sequence[end]
            start -= 1
            end -= 1
            prefix_sum += sequence[start]
        else:
            while start != 0 and sequence[start - 1] == sequence[end]:
                start -= 1
                end -= 1
            break

    return [start, end]
```
