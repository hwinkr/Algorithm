```py
def count_one(n:int) -> int:
    count = 0
    while n > 0:
        if n % 2:
            count += 1
        n //= 2
    return count

def solution(n):
    one = count_one(n)

    answer = n + 1
    answer_one = count_one(answer)

    while answer_one != one:
        answer += 1
        answer_one = count_one(answer)

    return answer
```
