def solution(arr1, arr2):
    answer = []

    row, col = len(arr1), len(arr2[0])
    common = len(arr1[0])

    for i in range(row):
        tmp_arr = []
        for j in range(col):
            tmp_number = 0
            for k in range(common):
                tmp_number += arr1[i][k] * arr2[k][j]
            tmp_arr.append(tmp_number)
        answer.append(tmp_arr)

    return answer
