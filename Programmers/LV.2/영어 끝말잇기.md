def solution(n, words):
    people = [i + 1 for i in range(n)]
    valid_words = []

    for i, word in enumerate(words):
        if not valid_words:
            valid_words.append(word)
            continue

        if word[0] != valid_words[-1][-1] or word in valid_words or len(word) == 1:
            return [people[i % n], i // n + 1]
        valid_words.append(word)

    return [0, 0]
