def solution(n, m, section):
    answer = 0

    while section:
        if section[-1] - m < 1:
            for i in range(section[-1], 0, -1):
                if section and i == section[-1]:
                    section.pop()
        else:
            for i in range(section[-1], section[-1] - m, -1):
                if section and i == section[-1]:
                    section.pop()

        answer += 1

    return answer


print(solution(4, 1, [1, 2, 3, 4]))
