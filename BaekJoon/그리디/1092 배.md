import sys

input = sys.stdin.readline


def sol(limits: list, weights: list) -> int:
    time = 0

    if weights[0] > limits[0]:
        return -1
    else:
        while weights:
            for limit in limits:
                for weight in weights:
                    if limit >= weight:
                        weights.remove(weight)
                        break
            time += 1

    return time


if __name__ == "__main__":
    n = int(input())
